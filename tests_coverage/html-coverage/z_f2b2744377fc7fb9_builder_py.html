28@dataclass 

29class Builder: 

30 """ 

31 The Builder is an auxiliary class used to build portfolios. 

32 It overloads the __iter__ method to allow the class to iterate over 

33 the timestamps for which the portfolio data is available. 

34 

35 In each iteration we can update the portfolio by setting either 

36 the weights, the position or the cash position. 

37 

38 After the iteration has been completed we build a Portfolio object 

39 by calling the build method. 

40 """ 

41 

42 prices: pd.DataFrame 

43 initial_aum: float = 1e6 

44 

45 _state: State = None 

46 _units: pd.DataFrame = None 

47 _aum: pd.Series = None 

48 

49 def __post_init__(self) -> None: 

50 """ 

51 The __post_init__ method is a special method of initialized instances 

52 of the _Builder class and is called after initialization. 

53 It sets the initial amount of cash in the portfolio to be equal to the input initial_cash parameter. 

54 

55 The method takes no input parameter. It initializes the cash attribute in the internal 

56 _State object with the initial amount of cash in the portfolio, self.initial_cash. 

57 

58 Note that this method is often used in Python classes for additional initialization routines 

59 that can only be performed after the object is fully initialized. __post_init__ 

60 is called automatically after the object initialization. 

61 """ 

62 

63 # assert isinstance(self.prices, pd.DataFrame) 

64 assert self.prices.index.is_monotonic_increasing 

65 assert self.prices.index.is_unique 

66 

67 self._state = State() 

68 

69 self._units = pd.DataFrame( 

70 index=self.prices.index, 

71 columns=self.prices.columns, 

72 data=np.nan, 

73 dtype=float, 

74 ) 

75 

76 self._aum = pd.Series(index=self.prices.index, dtype=float) 

77 

78 self._state.aum = self.initial_aum 

79 

80 @property 

81 def valid(self): 

82 """ 

83 Analyse the validity of the data 

84 Do it column by column of the prices 

85 """ 

86 return self.prices.apply(valid) 

87 

88 @property 

89 def intervals(self): 

90 """ 

91 Find for each column the first and the last valid index 

92 """ 

93 return self.prices.apply( 

94 lambda ts: pd.Series( 

95 {"first": ts.first_valid_index(), "last": ts.last_valid_index()} 

96 ) 

97 ).transpose() 

98 

99 @property 

100 def index(self) -> pd.DatetimeIndex: 

101 """A property that returns the index of the portfolio, 

102 which are the timestamps for which the portfolio data is available. 

103 

104 Returns: pd.Index: A pandas index representing the 

105 time period for which the portfolio data is available. 

106 """ 

107 return pd.DatetimeIndex(self.prices.index) 

108 

109 @property 

110 def current_prices(self) -> np.array: 

111 """ 

112 Get the current prices from the state 

113 """ 

114 return self._state.prices[self._state.assets].values 

115 

116 def __iter__(self) -> Generator[tuple[pd.DatetimeIndex, State], None, None]: 

117 """ 

118 The __iter__ method allows the object to be iterated over in a for loop, 

119 yielding time and the current state of the portfolio. 

120 The method yields a list of dates seen so far and returns a tuple 

121 containing the list of dates and the current portfolio state. 

122 

123 Yield: 

124 

125 time: a pandas DatetimeIndex object containing the dates seen so far. 

126 state: the current state of the portfolio, 

127 

128 taking into account the stock prices at each interval. 

129 """ 

130 for t in self.index: 

131 # update the current prices for the portfolio 

132 self._state.prices = self.prices.loc[t] 

133 

134 # update the current time for the state 

135 self._state.time = t 

136 

137 # yield the vector of times seen so far and the current state 

138 yield self.index[self.index <= t], self._state 

139 

140 @property 

141 def position(self) -> pd.Series: 

142 """ 

143 The position property returns the current position of the portfolio. 

144 It returns a pandas Series object containing the current position of the portfolio. 

145 

146 Returns: pd.Series: a pandas Series object containing the current position of the portfolio. 

147 """ 

148 return self._units.loc[self._state.time] 

149 

150 @position.setter 

151 def position(self, position: pd.Series) -> None: 

152 """ 

153 The position property returns the current position of the portfolio. 

154 It returns a pandas Series object containing the current position of the portfolio. 

155 

156 Returns: pd.Series: a pandas Series object containing the current position of the portfolio. 

157 """ 

158 self._units.loc[self._state.time, self._state.assets] = position 

159 self._state.position = position 

160 

161 @property 

162 def cashposition(self): 

163 """ 

164 The cashposition property returns the current cash position of the portfolio. 

165 """ 

166 return self.position * self.current_prices 

167 

168 @property 

169 def units(self): 

170 """ 

171 The units property returns the frame of holdings of the portfolio. 

172 Useful mainly for testing 

173 """ 

174 return self._units 

175 

176 @cashposition.setter 

177 def cashposition(self, cashposition: pd.Series) -> None: 

178 """ 

179 The cashposition property sets the current cash position of the portfolio. 

180 """ 

181 self.position = cashposition / self.current_prices 

182 

183 def build(self): 

184 """A function that creates a new instance of the EquityPortfolio 

185 class based on the internal state of the Portfolio builder object. 

186 

187 Returns: EquityPortfolio: A new instance of the EquityPortfolio class 

188 with the attributes (prices, units, initial_cash, trading_cost_model) as specified in the Portfolio builder. 

189 

190 Notes: The function simply creates a new instance of the EquityPortfolio 

191 class with the attributes (prices, units, initial_cash, trading_cost_model) equal 

192 to the corresponding attributes in the Portfolio builder object. 

193 The resulting EquityPortfolio object will have the same state as the Portfolio builder from which it was built. 

194 """ 

195 

196 return Portfolio(prices=self.prices, units=self.units, aum=self.aum) 

197 

198 @property 

199 def weights(self) -> np.array: 

200 """ 

201 Get the current weights from the state 

202 """ 

203 return self._state.weights[self._state.assets].values 

204 

205 @weights.setter 

206 def weights(self, weights: np.array) -> None: 

207 """ 

208 The weights property sets the current weights of the portfolio. 

209 We convert the weights to positions using the current prices and the NAV 

210 """ 

211 self.position = self._state.nav * weights / self.current_prices 

212 

213 @property 

214 def aum(self): 

215 """ 

216 The aum property returns the current AUM of the portfolio. 

217 """ 

218 return self._aum 

219 

220 @aum.setter 

221 def aum(self, aum): 

222 """ 

223 The aum property sets the current AUM of the portfolio. 

224 """ 

225 self._aum[self._state.time] = aum 

226 self._state.aum = aum

228 @classmethod 

229 def from_returns(cls, returns): 

230 """Build Futures Portfolio from returns""" 

231 # compute artificial prices (but scaled such their returns are correct) 

232 

233 prices = returns2prices(returns) 

234 return cls(prices=prices)
